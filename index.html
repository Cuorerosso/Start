<script>

    // TRADUZIONI

    const texts = {

        it: {

            homeTitle: "Benvenuto nel mio sito!",

            page1Title: "📖 Pagina 1",

            page2Title: "🎨 Pagina 2",

            page2Text: "Questo è il contenuto della seconda pagina. Personalizzala come preferisci!",

            page3Title: "🎵 Pagina 3",

            page3Text: "Questo è il contenuto della terza pagina. Aggiungi immagini, testo o quello che vuoi!",

            page4Title: "📞 Pagina 4",

            page4Text: "Questo è il contenuto della quarta pagina. Personalizzala con le tue informazioni!",

            homeBtn: "🏠 Torna alla Home"

        },

        en: {

            homeTitle: "Welcome to my site!",

            homeSubtitle: "Click a button to explore",

            page1Title: "📖 Page 1",

            page1Text: "This is the content of the first page. You can write whatever you want here!",

            page2Title: "🎨 Page 2",

            page2Text: "This is the content of the second page. Customize it as you like!",

            page3Title: "🎵 Page 3",

            page3Text: "This is the content of the third page. Add images, text, or anything you want!",

            page4Title: "📞 Page 4",

            page4Text: "This is the content of the fourth page. Customize it with your information!",

            homeBtn: "🏠 Back to Home"

        }

    };


    let currentLang = 'it';


    function updateLanguage() {

        const t = texts[currentLang];

        document.getElementById('homeTitle').innerText = t.homeTitle;

        document.getElementById('homeSubtitle').innerText = t.homeSubtitle;

        document.getElementById('page1Title').innerText = t.page1Title;

        document.getElementById('page1Text').innerText = t.page1Text;

        document.getElementById('page2Title').innerText = t.page2Title;

        document.getElementById('page2Text').innerText = t.page2Text;

        document.getElementById('page3Title').innerText = t.page3Title;

        document.getElementById('page3Text').innerText = t.page3Text;

        document.getElementById('page4Title').innerText = t.page4Title;

        document.getElementById('page4Text').innerText = t.page4Text;

        

        // Aggiorna anche i pulsanti "Torna alla Home" in tutte le pagine

        const homeBtns = document.querySelectorAll('.btn-home');

        homeBtns.forEach(btn => {

            btn.innerText = t.homeBtn;

        });

        

        document.getElementById('flagBtn').innerText = currentLang === 'it' ? '🇬🇧' : '🇮🇹';

    }


    function toggleLanguage() {

        currentLang = currentLang === 'it' ? 'en' : 'it';

        updateLanguage();

        localStorage.setItem('lang', currentLang);

    }


    function showPage(pageNum) {

        document.getElementById('homePage').style.display = 'none';

        for(let i = 1; i <= 4; i++) {

            const page = document.getElementById('page' + i);

            if(page) page.classList.remove('active');

        }

        const selectedPage = document.getElementById('page' + pageNum);

        selectedPage.classList.add('active');

        selectedPage.style.display = 'block';

    }


    function showHome() {

        for(let i = 1; i <= 4; i++) {

            const page = document.getElementById('page' + i);

            page.classList.remove('active');

            page.style.display = 'none';

        }

        document.getElementById('homePage').style.display = 'block';

    }


    // All'avvio

    document.addEventListener('DOMContentLoaded', () => {

        const savedLang = localStorage.getItem('lang');

        if(savedLang === 'en' || savedLang === 'it') {

            currentLang = savedLang;

        }

        updateLanguage();

        showHome();

    });
